sudo mkdir /media/mydisk

sudo mkfs.ext4 /dev/sdb1

sudo mount /dev/sdb1  /media/mydisk

# Managing user and Groups

4 main user administration files

/etc/passwd : user account and password info

/etc/shadow : Hold the encrypted pw of the corresponding account

/etc/group  : group information for each account

/etc/gshadow : The file contains secure group account information
